// ── DATA ───────────────────────────────────────────────
const STORAGE_KEY = 'faf_foia_cases';
let cases = JSON.parse(localStorage.getItem(STORAGE_KEY) || '[]');
let editingId = null;

function save() {
  localStorage.setItem(STORAGE_KEY, JSON.stringify(cases));
}

function genId() {
  const num = String(cases.length + 1).padStart(4, '0');
  return `FAF-${num}`;
}

function today() {
  return new Date().toISOString().split('T')[0];
}

function fmtDate(d) {
  if (!d) return '—';
  const [y, m, day] = d.split('-');
  return `${m}/${day}/${y}`;
}

function daysUntil(dateStr) {
  if (!dateStr) return null;
  const diff = new Date(dateStr) - new Date(today());
  return Math.ceil(diff / 86400000);
}

function statusColor(status) {
  const colors = {
    pending: 'var(--orange)',
    submitted: 'var(--blue)',
    received: 'var(--green)',
    denied: 'var(--red)',
    appealing: 'var(--purple)',
    delayed: 'var(--accent)',
    closed: 'var(--muted)',
  };
  return colors[status] || 'var(--muted)';
}

function toast(msg) {
  const el = document.getElementById('toast');
  el.textContent = msg;
  el.classList.add('show');
  setTimeout(() => el.classList.remove('show'), 2500);
}

// ── RENDER ALL ─────────────────────────────────────────
function renderAll() {
  renderHeader();
  renderDashboard();
  renderCases();
  renderDeadlines();
  populateStateFilter();
}

function renderHeader() {
  document.getElementById('hdr-total').textContent = cases.length;
  document.getElementById('hdr-pending').textContent = cases.filter(c => ['pending','submitted','delayed'].includes(c.status)).length;
  document.getElementById('hdr-received').textContent = cases.filter(c => c.status === 'received').length;
}

// ── DASHBOARD ──────────────────────────────────────────
function renderDashboard() {
  const total = cases.length;
  const submitted = cases.filter(c => c.status === 'submitted').length;
  const received  = cases.filter(c => c.status === 'received').length;
  const denied    = cases.filter(c => c.status === 'denied').length;
  const pending   = cases.filter(c => c.status === 'pending').length;
  const delayed   = cases.filter(c => c.status === 'delayed').length;
  const overdue   = cases.filter(c => c.dueDate && daysUntil(c.dueDate) < 0 && !['received','closed','denied'].includes(c.status)).length;

  const kpiEl = document.getElementById('dash-kpis');
  kpiEl.innerHTML = [
    { label: 'Total Requests', value: total, color: 'var(--accent)' },
    { label: 'Pending Send',   value: pending, color: 'var(--orange)' },
    { label: 'Submitted',      value: submitted, color: 'var(--blue)' },
    { label: 'Received',       value: received, color: 'var(--green)' },
    { label: 'Denied',         value: denied, color: 'var(--red)' },
    { label: 'Delayed',        value: delayed, color: 'var(--purple)' },
    { label: 'Overdue',        value: overdue, color: overdue > 0 ? 'var(--red)' : 'var(--muted)' },
    { label: 'Ready to Edit',  value: cases.filter(c => c.ready === 'yes').length, color: 'var(--green)' },
  ].map(k => `
    <div class="kpi-card">
      <div class="kpi-value" style="color:${k.color}">${k.value}</div>
      <div class="kpi-label">${k.label}</div>
    </div>
  `).join('');

  // Recent
  const recentEl = document.getElementById('dash-recent');
  const recent = [...cases].reverse().slice(0, 6);
  if (!recent.length) {
    recentEl.innerHTML = '<div class="empty-state"><div class="empty-icon">📂</div>No cases yet</div>';
  } else {
    recentEl.innerHTML = recent.map(c => `
      <div class="recent-item" onclick="openDetail('${c.id}')">
        <div>
          <div style="font-family:var(--mono);font-size:11px;color:var(--accent)">${c.id}</div>
          <div style="font-size:13px;margin-top:2px">${c.name}</div>
          <div style="font-family:var(--mono);font-size:11px;color:var(--muted)">${c.agency} • ${c.state || '—'}</div>
        </div>
        <span class="tag tag-status-${c.status}">${c.status}</span>
      </div>
    `).join('');
  }

  // Deadlines
  const ddEl = document.getElementById('dash-deadlines');
  const upcoming = cases
    .filter(c => c.dueDate && !['received','closed'].includes(c.status))
    .map(c => ({ ...c, days: daysUntil(c.dueDate) }))
    .sort((a, b) => a.days - b.days)
    .slice(0, 6);

  if (!upcoming.length) {
    ddEl.innerHTML = '<div style="color:var(--muted);font-family:var(--mono);font-size:12px;padding:12px 0">No upcoming deadlines</div>';
  } else {
    ddEl.innerHTML = upcoming.map(c => {
      const overdue = c.days < 0;
      const label = overdue ? `${Math.abs(c.days)}d OVERDUE` : c.days === 0 ? 'DUE TODAY' : `${c.days}d left`;
      return `
        <div class="deadline-item" onclick="openDetail('${c.id}')">
          <div>
            <div style="font-family:var(--mono);font-size:11px;color:var(--accent)">${c.id}</div>
            <div style="font-size:13px">${c.name}</div>
          </div>
          <div style="text-align:right">
            <div style="font-family:var(--mono);font-size:11px;color:${overdue ? 'var(--red)' : c.days <= 3 ? 'var(--orange)' : 'var(--muted)'};font-weight:600">${label}</div>
            <div style="font-family:var(--mono);font-size:10px;color:var(--muted)">${fmtDate(c.dueDate)}</div>
          </div>
        </div>
      `;
    }).join('');
  }
}

// ── CASES LIST ─────────────────────────────────────────
function populateStateFilter() {
  const states = [...new Set(cases.map(c => c.state).filter(Boolean))].sort();
  const sel = document.getElementById('filter-state');
  const cur = sel.value;
  sel.innerHTML = '<option value="">All States</option>' + states.map(s => `<option value="${s}" ${s===cur?'selected':''}>${s}</option>`).join('');
}

function renderCases() {
  const search = (document.getElementById('search-input').value || '').toLowerCase();
  const statusF = document.getElementById('filter-status').value;
  const stateF  = document.getElementById('filter-state').value;

  let filtered = cases.filter(c => {
    if (statusF && c.status !== statusF) return false;
    if (stateF  && c.state  !== stateF)  return false;
    if (search) {
      const hay = `${c.name} ${c.agency} ${c.state} ${c.notes} ${c.request} ${c.refnum}`.toLowerCase();
      if (!hay.includes(search)) return false;
    }
    return true;
  }).reverse();

  const el = document.getElementById('cases-list');
  if (!filtered.length) {
    el.innerHTML = cases.length
      ? '<div class="no-results">No cases match your filters.</div>'
      : '<div class="empty-state"><div class="empty-icon">🗂️</div>No FOIA requests yet.<br>Hit "+ New Case" to log your first one.</div>';
    return;
  }

  el.innerHTML = filtered.map(c => {
    const days = c.dueDate ? daysUntil(c.dueDate) : null;
    const overdue = days !== null && days < 0 && !['received','closed','denied'].includes(c.status);
    let dueLabel = '';
    if (c.dueDate && !['received','closed'].includes(c.status)) {
      if (days < 0) dueLabel = `<span class="case-date overdue">⚠ ${Math.abs(days)}d overdue</span>`;
      else if (days === 0) dueLabel = `<span class="case-date overdue">⚠ Due today</span>`;
      else if (days <= 5) dueLabel = `<span class="case-date" style="color:var(--orange)">${days}d left</span>`;
      else dueLabel = `<span class="case-date">Due ${fmtDate(c.dueDate)}</span>`;
    }

    const potentialEmoji = { high: '🔥', medium: '📈', low: '📉' }[c.potential] || '';

    return `
      <div class="case-card" style="--status-color:${statusColor(c.status)}" onclick="openDetail('${c.id}')">
        <div style="display:flex;justify-content:space-between;align-items:flex-start;gap:12px;">
          <div style="flex:1">
            <div style="display:flex;align-items:center;gap:8px;margin-bottom:4px;">
              <span class="case-id">${c.id}</span>
              ${c.refnum ? `<span style="font-family:var(--mono);font-size:10px;color:var(--muted)">REF: ${c.refnum}</span>` : ''}
            </div>
            <div class="case-title">${c.name} ${potentialEmoji}</div>
            <div class="case-sub">${c.agency}${c.state ? ' · ' + c.state : ''}</div>
            <div class="case-meta">
              <span class="tag tag-status-${c.status}">${c.status}</span>
              ${c.state ? `<span class="tag tag-state">${c.state}</span>` : ''}
              ${c.ready === 'yes' ? `<span class="tag" style="background:rgba(39,174,96,0.15);color:var(--green)">READY</span>` : ''}
              ${dueLabel}
              ${c.submittedDate ? `<span class="case-date">Filed ${fmtDate(c.submittedDate)}</span>` : ''}
            </div>
          </div>
          <button class="btn btn-ghost btn-sm" onclick="event.stopPropagation();openEdit('${c.id}')">Edit</button>
        </div>
      </div>
    `;
  }).join('');
}

document.getElementById('search-input').addEventListener('input', renderCases);
document.getElementById('filter-status').addEventListener('change', renderCases);
document.getElementById('filter-state').addEventListener('change', renderCases);

// ── DEADLINES VIEW ─────────────────────────────────────
function renderDeadlines() {
  const all = cases
    .filter(c => c.dueDate)
    .map(c => ({ ...c, days: daysUntil(c.dueDate) }))
    .sort((a, b) => a.days - b.days);

  const el = document.getElementById('deadlines-list');
  if (!all.length) {
    el.innerHTML = '<div class="empty-state"><div class="empty-icon">📅</div>No deadlines set yet.</div>';
    return;
  }

  el.innerHTML = all.map(c => {
    const done = ['received','closed','denied'].includes(c.status);
    const overdue = !done && c.days < 0;
    const label = done ? '✓ Done' : overdue ? `${Math.abs(c.days)}d OVERDUE` : c.days === 0 ? 'DUE TODAY' : `${c.days}d remaining`;
    const color = done ? 'var(--muted)' : overdue ? 'var(--red)' : c.days <= 3 ? 'var(--orange)' : 'var(--green)';

    return `
      <div class="deadline-item" onclick="openDetail('${c.id}')">
        <div>
          <div style="font-family:var(--mono);font-size:11px;color:var(--accent)">${c.id}</div>
          <div style="font-size:14px;font-weight:500">${c.name}</div>
          <div style="font-family:var(--mono);font-size:11px;color:var(--muted)">${c.agency} · ${c.state || '—'}</div>
        </div>
        <div style="text-align:right;flex-shrink:0">
          <div style="font-family:var(--mono);font-size:12px;font-weight:700;color:${color}">${label}</div>
          <div style="font-family:var(--mono);font-size:11px;color:var(--muted);margin-top:2px">${fmtDate(c.dueDate)}</div>
          <span class="tag tag-status-${c.status}" style="margin-top:4px;display:inline-block">${c.status}</span>
        </div>
      </div>
    `;
  }).join('');
}

function openAddModal() {
  editingId = null;
  document.getElementById('modal-mode-label').textContent = 'New Case';
  document.getElementById('delete-case-btn').style.display = 'none';
  clearForm();
  document.getElementById('f-submitted').value = today();
  document.getElementById('case-modal').classList.add('open');
}

function openEdit(id) {
  const c = cases.find(x => x.id === id);
  if (!c) return;
  editingId = id;
  document.getElementById('modal-mode-label').textContent = `Edit Case — ${id}`;
  document.getElementById('delete-case-btn').style.display = '';
  document.getElementById('f-name').value = c.name || '';
  document.getElementById('f-agency').value = c.agency || '';
  document.getElementById('f-state').value = c.state || '';
  document.getElementById('f-status').value = c.status || 'pending';
  document.getElementById('f-submitted').value = c.submittedDate || '';
  document.getElementById('f-due').value = c.dueDate || '';
  document.getElementById('f-portal').value = c.portal || '';
  document.getElementById('f-refnum').value = c.refnum || '';
  document.getElementById('f-request').value = c.request || '';
  document.getElementById('f-notes').value = c.notes || '';
  document.getElementById('f-potential').value = c.potential || '';
  document.getElementById('f-ready').value = c.ready || 'no';
  document.getElementById('case-modal').classList.add('open');
}

function closeModal() {
  document.getElementById('case-modal').classList.remove('open');
}

function clearForm() {
  ['f-name','f-agency','f-state','f-submitted','f-due','f-portal','f-refnum','f-request','f-notes'].forEach(id => {
    document.getElementById(id).value = '';
  });
  document.getElementById('f-status').value = 'pending';
  document.getElementById('f-potential').value = '';
  document.getElementById('f-ready').value = 'no';
}

document.getElementById('modal-save').addEventListener('click', () => {
  const name   = document.getElementById('f-name').value.trim();
  const agency = document.getElementById('f-agency').value.trim();
  if (!name || !agency) { toast('Case name and agency are required.'); return; }

  if (editingId) {
    const idx = cases.findIndex(c => c.id === editingId);
    cases[idx] = {
      ...cases[idx],
      name, agency,
      state: document.getElementById('f-state').value.trim().toUpperCase(),
      status: document.getElementById('f-status').value,
      submittedDate: document.getElementById('f-submitted').value,
      dueDate: document.getElementById('f-due').value,
      portal: document.getElementById('f-portal').value.trim(),
      refnum: document.getElementById('f-refnum').value.trim(),
      request: document.getElementById('f-request').value.trim(),
      notes: document.getElementById('f-notes').value.trim(),
      potential: document.getElementById('f-potential').value,
      ready: document.getElementById('f-ready').value,
      updatedAt: new Date().toISOString(),
    };
    toast('Case updated.');
  } else {
    const c = {
      id: genId(),
      name, agency,
      state: document.getElementById('f-state').value.trim().toUpperCase(),
      status: document.getElementById('f-status').value,
      submittedDate: document.getElementById('f-submitted').value,
      dueDate: document.getElementById('f-due').value,
      portal: document.getElementById('f-portal').value.trim(),
      refnum: document.getElementById('f-refnum').value.trim(),
      request: document.getElementById('f-request').value.trim(),
      notes: document.getElementById('f-notes').value.trim(),
      potential: document.getElementById('f-potential').value,
      ready: document.getElementById('f-ready').value,
      timeline: [],
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString(),
    };
    cases.push(c);
    toast(`Case ${c.id} created.`);
  }
  save();
  closeModal();
  renderAll();
});

document.getElementById('delete-case-btn').addEventListener('click', () => {
  if (!editingId) return;
  if (!confirm('Delete this case permanently?')) return;
  cases = cases.filter(c => c.id !== editingId);
  save();
  closeModal();
  renderAll();
  toast('Case deleted.');
});

function openDetail(id) {
  const c = cases.find(x => x.id === id);
  if (!c) return;

  document.getElementById('detail-id').textContent = c.id;
  document.getElementById('detail-name').textContent = c.name;

  document.getElementById('detail-body').innerHTML = `

      <div class="detail-section-title">Request Info</div>
      <div class="detail-grid">
        <div class="detail-item"><div class="detail-label">Agency</div><div class="detail-value">${c.agency}</div></div>

      <div class="note-input-row">

function addNote(id) {
  const input = document.getElementById('note-input');
  const text = input.value.trim();
  if (!text) return;
  const c = cases.find(x => x.id === id);
  if (!c) return;
  if (!c.timeline) c.timeline = [];
  c.timeline.push({ ts: new Date().toISOString(), text });
  save();
  input.value = '';
  const list = document.getElementById('timeline-list');
  list.innerHTML = c.timeline.map(n => `
    <div class="note-item">
      <div class="note-ts">${new Date(n.ts).toLocaleString()}</div>
      <div class="note-text">${n.text}</div>
    </div>
  `).join('');
  renderAll();
}

function quickStatus(id, val) {
  if (!val) return;
  const c = cases.find(x => x.id === id);
  if (!c) return;
  c.status = val;
  c.updatedAt = new Date().toISOString();
  if (!c.timeline) c.timeline = [];
  c.timeline.push({ ts: new Date().toISOString(), text: `Status changed to: ${val}` });
  save();
  renderAll();
  toast(`Status → ${val}`);
  document.getElementById(`quick-status-${id}`).value = '';
}
